def read_rss(source, show_count = 3)
  @inhibit_block_run = true
  rss, = Rss.new
  rss.read_rss(source, show_count)
  loop do
    article = rss.next_article
    return unless article
    test_words = TestWords.new
    @words.assign article
    test_words.run @words
  end
end

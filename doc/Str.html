def to_s
  delim  = Params.delim_str
  [
    Params.name + "\n",
    delim,
    Params.wpm_str,
    Params.shuffle_str,
    Params.word_count_str,
    Params.word_size_str,
    beginning_str,
    ending_str,
    delim
  ].collect{ |prm| prm.to_s }.join
end

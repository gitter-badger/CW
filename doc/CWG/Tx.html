def echo
  loop do
    @temp ||= ''
    if @temp != ''
      puts "@temp = #{@temp}"
      chr = @temp.shift
      puts "chr = #{chr}"
      @audio.convert_words(chr)
      @audio.play
    end
    sleep 0.1
  end
end

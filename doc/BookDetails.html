def sentences_complete?
  if @args.has_key?(:sentences) &&  @args[:sentences].is_a?(Fixnum)
    if @sentence_count_source
      @sentence_count_source = nil
    else
      @args[:sentences] -= 1
      @sentence_count_source = true
    end
    true if(@args[:sentences] < 0)
  end
end

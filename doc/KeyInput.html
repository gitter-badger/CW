def quit_input?
  @quit_ary.rotate!
  @quit_ary[3] = @chr
  quit_str = @quit_ary.join.downcase
  if quit_str == 'qqqq'
    reset_stdin
    return true
  end
end

def read
  @chr = nil
  begin
    system("stty raw -echo")
    @chr = STDIN.getc
  ensure
    system("stty raw -echo")
  end
end

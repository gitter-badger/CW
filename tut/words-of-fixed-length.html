Words of Fixed Length

1) Write a test that generates 5 words of 4 characters (test-by-letter)


# five-words-five-letters.rb

require 'cw'

CW.new do
  comment "5 words having 4 letters (12 WPM)"
  shuffle
  word_count    5
  word_length   4
  wpm           12
  test_letters
end

(download: five-words-five-letters.rb)

2) Write a test that generates 4 words of 6 characters


# five-words-five-letters.rb

require 'cw'

CW.new do
  comment "4 words having 6 letters (15 WPM)"
  shuffle
  wpm                 15
  number_of_words     4   # you can use number_of_words or word_count
  word_having_size_of 6   # you can use word_having_size_of or word_length
end

(download: four-words-six-letters.rb)
